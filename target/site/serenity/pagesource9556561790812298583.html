	function actualizaAlarma(id, id_spot, modo, element){

		$(element).empty();

		var params = "id_alarma="+id;

		A.enviarPostDatos('/privada/actualiza_alarma', params, function(res) {

			$('#burbuja-alarma')[0].innerText = (parseInt($('#burbuja-alarma')[0].innerText)-1);

			abreDetalle(id_spot, modo);

		});

	}

	

	function actualizaAlarmaProveedor(id, id_proveedor, nombre_proveedor, element){

		$(element).empty();

		var params = "id_alarma="+id;

		A.enviarPostDatos('/privada/actualiza_alarma', params, function(res) {

			$('#burbuja-alarma')[0].innerText = (parseInt($('#burbuja-alarma')[0].innerText)-1);

			abreDetalleProveedor(id_proveedor, nombre_proveedor);

		});

	}



	function pintarAlarmas(){

		var params = '';

		A.enviarPostDatos('/privada/trae_alarmas', params, function(res) {

			const alarmas_nuevas = JSON.parse(res);

			if(alarmas_nuevas.length == 0){

				if($('#alarmasC').hasClass('pM')){

					$('#alarmasC').removeClass('pM');

				}



				if($('#alarmasC').hasClass('dropdown')){

					$('#alarmasC').removeClass('dropdown');

				}				



				if($('#alarmas').hasClass('fa-bell')){

					$('#alarmas').removeClass('fa-bell');

				}



				if($('#alarmas').hasClass('fa-bell-o') == false){

					$('#alarmas').addClass('fa-bell-o');

				}



				if($('#burbuja-alarma').hasClass('no_alarma') == false){

					$('#burbuja-alarma').addClass('no_alarma');

				}



				$('#burbuja-alarma').empty();



				$('#menu-alarmas').empty();

			} else{

				if($('#alarmasC').hasClass('pM') == false){

					$('#alarmasC').addClass('pM');

				}



				if($('#alarmasC').hasClass('dropdown') == false){

					$('#alarmasC').addClass('dropdown');

				}



				if($('#alarmas').hasClass('fa-bell') == false){

					$('#alarmas').addClass('fa-bell');

				}



				if($('#alarmas').hasClass('fa-bell-o')){

					$('#alarmas').removeClass('fa-bell-o');

				}



				if($('#burbuja-alarma').hasClass('no_alarma')){

					$('#burbuja-alarma').removeClass('no_alarma');

				}



				$('#burbuja-alarma')[0].innerText = alarmas_nuevas.length;



				$('#menu-alarmas').empty();

			}

			for(var i = 0; i < alarmas_nuevas.length; i++){

				switch(alarmas_nuevas[i].tipo) {

					case 1:

						$('#menu-alarmas').append('<li onclick="actualizaAlarma('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_spot+', '+alarmas_nuevas[i].tipo_spot+', this);">Nueva oferta en el Tender Spot '+alarmas_nuevas[i].id_spot+'.</li>')

						break;

					case 2:

						$('#menu-alarmas').append('<li onclick="actualizaAlarma('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_spot+', '+alarmas_nuevas[i].tipo_spot+', this);">Oferta modificada en el Tender Spot '+alarmas_nuevas[i].id_spot+'.</li>');

						break;

					case 3:

						$('#menu-alarmas').append('<li onclick="actualizaAlarma('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_spot+', '+alarmas_nuevas[i].tipo_spot+', this);">Un proveedor ha rechazado la participación en el Tender Spot '+alarmas_nuevas[i].id_spot+'.</li>');

						break;

					case 4:

						$('#menu-alarmas').append('<li onclick="actualizaAlarma('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_spot+', '+alarmas_nuevas[i].tipo_spot+', this);">Fallo al enviar la invitación del Tender Spot '+alarmas_nuevas[i].id_spot+'.</li>');

						break;

					case 5:

						$('#menu-alarmas').append('<li onclick="actualizaAlarma('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_spot+', '+alarmas_nuevas[i].tipo_spot+', this);">El plazo de respuesta del Tender Spot '+alarmas_nuevas[i].id_spot+' ha finalizado.</li>');

						break;

					case 6:

						$('#menu-alarmas').append('<li onclick="actualizaAlarmaProveedor('+alarmas_nuevas[i].id+', '+alarmas_nuevas[i].id_proveedor+', \''+alarmas_nuevas[i].nombre_proveedor+'\', this);">Nueva solicitud del Proveedor: '+alarmas_nuevas[i].nombre_proveedor+'.</li>');

						break;

				}

			}

		});		

	}



	function abreDetalle(id_spot, modo){



		if(modo == 1) {

			if(parent.WM){

				parent.window.WM.abrirVentana('Tender Spot '+id_spot,'/modulos/tender_spot/detalle_spot_terrestre/?id_spot='+id_spot, false);

			}else{

				window.open('/modulos/tender_spot/detalle_spot');

			}

		} else if(modo == 2){

			if(parent.WM){

				parent.window.WM.abrirVentana('Tender Spot '+id_spot,'/modulos/tender_spot/detalle_spot/?id_spot='+id_spot, false);

			}else{

				window.open('/modulos/tender_spot/detalle_spot');

			}

		} else {

			if(parent.WM){

				parent.window.WM.abrirVentana('Tender Spot '+id_spot,'/modulos/tender_spot/detalle_spot_maritimo/?id_spot='+id_spot, false);

			}else{

				window.open('/modulos/tender_spot/detalle_spot');

			}

		}



	}

	}

	function escondeSubMenu(th){

		if(th.parentNode.children[1].style.display == "none"){

			th.parentNode.children[1].style.display = "";

			th.children[0].children[1].children[0].className = "fas fa-angle-right fa fa-2x"

		} else {

			th.parentNode.children[1].style.display = "none";

			th.children[0].children[1].children[0].className = "fas fa-angle-down fa fa-2x"

		}

	}

  var Accordion = function(el, multiple) {

    this.el = el || {};

    this.multiple = multiple || false;

      $this = $(this),

    $next.slideToggle();

    $this.parent().toggleClass('open');



    if (!e.data.multiple) {

      $el.find('.submenu').not($next).slideUp().parent().removeClass('open');

    };

  }

});
